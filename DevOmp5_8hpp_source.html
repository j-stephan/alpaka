57  //! The Omp5 device implementation.
58  class DevOmp5Impl
59  {
60  public:
61  //-----------------------------------------------------------------------------
62  DevOmp5Impl(int iDevice) noexcept : m_iDevice(iDevice) {}
63  //-----------------------------------------------------------------------------
64  DevOmp5Impl(DevOmp5Impl const &) = delete;
65  //-----------------------------------------------------------------------------
66  DevOmp5Impl(DevOmp5Impl &&) = delete;
67  //-----------------------------------------------------------------------------
68  auto operator=(DevOmp5Impl const &) -> DevOmp5Impl & = delete;
69  //-----------------------------------------------------------------------------
70  auto operator=(DevOmp5Impl &&) -> DevOmp5Impl & = delete;
71  //-----------------------------------------------------------------------------
72  ~DevOmp5Impl() = default;
73 
74  //-----------------------------------------------------------------------------
75  ALPAKA_FN_HOST auto getAllExistingQueues() const
76  -> std::vector<std::shared_ptr<queue::IGenericThreadsQueue<DevOmp5>>>
77  {
78  std::vector<std::shared_ptr<queue::IGenericThreadsQueue<DevOmp5>>> vspQueues;
79 
80  std::lock_guard<std::mutex> lk(m_Mutex);
81  vspQueues.reserve(m_queues.size());
82 
83  for(auto it = m_queues.begin(); it != m_queues.end();)
84  {
85  auto spQueue(it->lock());
86  if(spQueue)
87  {
88  vspQueues.emplace_back(std::move(spQueue));
89  ++it;
90  }
91  else
92  {
93  it = m_queues.erase(it);
94  }
95  }
96  return vspQueues;
97  }
98 
99  //-----------------------------------------------------------------------------
100  //! Registers the given queue on this device.
101  //! NOTE: Every queue has to be registered for correct functionality of device wait operations!
102  ALPAKA_FN_HOST auto registerQueue(std::shared_ptr<queue::IGenericThreadsQueue<DevOmp5>> spQueue)
103  -> void
104  {
105  std::lock_guard<std::mutex> lk(m_Mutex);
106 
107  // Register this queue on the device.
108  m_queues.push_back(spQueue);
109  }
110 
111  int iDevice() const {return m_iDevice;}
112 
113  private:
114  std::mutex mutable m_Mutex;
115  std::vector<std::weak_ptr<queue::IGenericThreadsQueue<DevOmp5>>> mutable m_queues;
116  const int m_iDevice;
117  };
118  }
