385  struct Print
386  {
387  ALPAKA_FN_HOST static auto print(
388  TView const & view,
389  Elem<TView> const * const ptr,
390  Vec<Dim<TView>, Idx<TView>> const & extent,
391  std::ostream & os,
392  std::string const & elementSeparator,
393  std::string const & rowSeparator,
394  std::string const & rowPrefix,
395  std::string const & rowSuffix)
396  -> void
397  {
398  os << rowPrefix;
399 
400  auto const pitch(getPitchBytes<TDim::value+1u>(view));
401  auto const lastIdx(extent[TDim::value]-1u);
402  for(auto i(decltype(lastIdx)(0)); i<=lastIdx ;++i)
403  {
404  Print<
405  DimInt<TDim::value+1u>,
406  TView>
407  ::print(
408  view,
409  reinterpret_cast<Elem<TView> const *>(reinterpret_cast<std::uint8_t const *>(ptr)+i*pitch),
410  extent,
411  os,
412  elementSeparator,
413  rowSeparator,
414  rowPrefix,
415  rowSuffix);
416 
417  // While we are not at the end of a row, add the row separator.
418  if(i != lastIdx)
419  {
420  os << rowSeparator;
421  }
422  }
423 
424  os << rowSuffix;
425  }
426  };

428  template<
429  typename TView>
430  struct Print<
431  DimInt<Dim<TView>::value-1u>,
432  TView>
433  {
434  ALPAKA_FN_HOST static auto print(
435  TView const & view,
436  Elem<TView> const * const ptr,
