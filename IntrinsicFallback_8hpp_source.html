22  template<
23  typename TValue>
24  static auto popcountFallback(TValue value)
25  -> std::int32_t
26  {
27  TValue count = 0;
28  while(value != 0)
29  {
30  count += value & 1u;

32  }
33  return static_cast<std::int32_t>(count);
