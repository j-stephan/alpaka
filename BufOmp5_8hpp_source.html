57  //! The OMP5 memory buffer detail.
58  template<
59  typename TElem,
60  typename TDim,
61  typename TIdx>
62  class BufOmp5Impl
63  {
64  static_assert(
65  !std::is_const<TElem>::value,
66  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const elements!");
67  static_assert(
68  !std::is_const<TIdx>::value,
69  "The idx type of the buffer can not be const!");
70  private:
71  using Elem = TElem;
72  using Dim = TDim;
73  //-----------------------------------------------------------------------------
74  //! Calculate the pitches purely from the extents.
75  template<
76  typename TExtent>
77  ALPAKA_FN_HOST static auto calculatePitchesFromExtents(
78  TExtent const & extent)
79  -> vec::Vec<TDim, TIdx>
80  {
81  vec::Vec<TDim, TIdx> pitchBytes(vec::Vec<TDim, TIdx>::all(0));
82  pitchBytes[TDim::value - 1u] = extent[TDim::value - 1u] * static_cast<TIdx>(sizeof(TElem));
83  for(TIdx i = TDim::value - 1u; i > static_cast<TIdx>(0u); --i)
84  {
85  pitchBytes[i-1] = extent[i-1] * pitchBytes[i];
86  }
87  return pitchBytes;
88  }
89 
90  public:
91  //-----------------------------------------------------------------------------
92  //! Constructor
93  template<
94  typename TExtent>
95  ALPAKA_FN_HOST BufOmp5Impl(
96  dev::DevOmp5 const & dev,
97  TElem * const pMem,
98  TExtent const & extent) :
99  m_dev(dev),
100  m_extentElements(extent::getExtentVecEnd<TDim>(extent)),
101  m_pitchBytes(calculatePitchesFromExtents(m_extentElements)),
102  m_pMem(pMem)
103  {
104  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
105 
106  static_assert(
107  TDim::value == dim::Dim<TExtent>::value,
108  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be identical!");
109  static_assert(
110  std::is_same<TIdx, idx::Idx<TExtent>>::value,
111  "The idx type of TExtent and the TIdx template parameter have to be identical!");
112  }
113 
114  public:
115  dev::DevOmp5 m_dev;
116  vec::Vec<TDim, TIdx> m_extentElements;
117  vec::Vec<TDim, TIdx> m_pitchBytes;
118  TElem* m_pMem;
119 
120  BufOmp5Impl(const BufOmp5Impl&) = delete;
121  BufOmp5Impl(BufOmp5Impl &&) = default;
122  BufOmp5Impl& operator=(const BufOmp5Impl&) = delete;
123  BufOmp5Impl& operator=(BufOmp5Impl&&) = default;
124  ~BufOmp5Impl()
125  {
126  omp_target_free(m_pMem, m_dev.m_spDevOmp5Impl->iDevice());
127  }
128  };
129  }
