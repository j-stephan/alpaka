22  //! The CPU threads accelerator atomic ops.
23  //
24  // Atomics can be used in the grids, blocks and threads hierarchy levels.
25  // Atomics are not guaranteed to be save between devices.
26  //
27  // \tparam THashTableSize size of the hash table to allow concurrency between
28  // atomics to different addresses
29  template<size_t THashTableSize>
30  class AtomicStdLibLock
31  {
32  public:

34  typename TAtomic,

41  static constexpr size_t nextPowerOf2(size_t const value, size_t const bit = 0u)
42  {
43  return value <= (static_cast<size_t>(1u) << bit) ?
44  (static_cast<size_t>(1u) << bit) : nextPowerOf2(value, bit + 1u);
45  }
46 
47  //-----------------------------------------------------------------------------
48  //! get a hash value of the pointer
49  //
50  // This is no perfect hash, there will be collisions if the size of pointer type
51  // is not a power of two.
52  template<typename TPtr>
53  static size_t hash(TPtr const * const ptr)
54  {
55  size_t const ptrAddr = reinterpret_cast< size_t >( ptr );
56  // using power of two for the next division will increase the performance
57  constexpr size_t typeSizePowerOf2 = nextPowerOf2(sizeof(TPtr));
58  // division removes the stride between indices
59  return (ptrAddr / typeSizePowerOf2);
60  }
