249  //! The CUDA/HIP RT blocking queue enqueue trait specialization.
250  template<
251  typename TTask>
252  struct Enqueue<
253  queue::QueueUniformCudaHipRtBlocking,
254  TTask>
255  {
256  //#############################################################################
257  enum class CallbackState
258  {
259  enqueued,
260  notified,
261  finished,
262  };
263 
264  //#############################################################################
265  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
266  {
267  std::mutex m_mutex;
268  std::condition_variable m_event;
269  CallbackState state = CallbackState::enqueued;
270  };
271 
272  //-----------------------------------------------------------------------------
273 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
274  static void CUDART_CB uniformCudaHipRtCallback(cudaStream_t /*queue*/, cudaError_t /*status*/, void *arg)
275 #else
276  static void HIPRT_CB uniformCudaHipRtCallback(hipStream_t /*queue*/, hipError_t /*status*/, void *arg)
277 #endif
278  {
279  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread has already finished.
280  const auto pCallbackSynchronizationData = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
281 
282  // Notify the executing thread.
283  {
284  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
285  pCallbackSynchronizationData->state = CallbackState::notified;
286  }
287  pCallbackSynchronizationData->m_event.notify_one();
288 
289  // Wait for the executing thread to finish the task if it has not already finished.
290  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
291  if(pCallbackSynchronizationData->state != CallbackState::finished)
292  {
293  pCallbackSynchronizationData->m_event.wait(
294  lock,
295  [pCallbackSynchronizationData](){
296  return pCallbackSynchronizationData->state == CallbackState::finished;
297  }
298  );
299  }
300  }
301 
302  //-----------------------------------------------------------------------------
303  ALPAKA_FN_HOST static auto enqueue(
304  queue::QueueUniformCudaHipRtBlocking & queue,
305  TTask const & task)
306  -> void
307  {
308 #if BOOST_COMP_HCC // NOTE: workaround for unwanted nonblocking hip streams for HCC (NVCC streams are blocking)
309  {
310  // thread-safe callee incrementing
311  std::lock_guard<std::mutex> guard(queue.m_spQueueImpl->m_mutex);
312  queue.m_spQueueImpl->m_callees += 1;
313  }
314 #endif
315  auto pCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
316 
317 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
318  ALPAKA_CUDA_RT_CHECK(cudaStreamAddCallback(
319  queue.m_spQueueImpl->m_UniformCudaHipQueue,
320  uniformCudaHipRtCallback,
321  pCallbackSynchronizationData.get(),
322  0u));
323 #else
324  ALPAKA_HIP_RT_CHECK(hipStreamAddCallback(
325  queue.m_spQueueImpl->m_UniformCudaHipQueue,
326  uniformCudaHipRtCallback,
327  pCallbackSynchronizationData.get(),
328  0u));
329 #endif
330  // We start a new std::thread which stores the task to be executed.
331  // This circumvents the limitation that it is not possible to call CUDA/HIP methods within the CUDA/HIP callback thread.
332  // The CUDA/HIP thread signals the std::thread when it is ready to execute the task.
333  // The CUDA/HIP thread is waiting for the std::thread to signal that it is finished executing the task
334  // before it executes the next task in the queue (CUDA/HIP stream).
335  std::thread t(
336  [pCallbackSynchronizationData,
337  task
338 #if BOOST_COMP_HCC // NOTE: workaround for unwanted nonblocking hip streams for HCC (NVCC streams are blocking)
339  ,&queue
340 #endif
341  ](){
342 
343 #if BOOST_COMP_HCC // NOTE: workaround for unwanted nonblocking hip streams for HCC (NVCC streams are blocking)
344  // thread-safe task execution and callee decrementing
345  std::lock_guard<std::mutex> guard(queue.m_spQueueImpl->m_mutex);
346 #endif
347 
348  // If the callback has not yet been called, we wait for it.
349  {
350  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
351  if(pCallbackSynchronizationData->state != CallbackState::notified)
352  {
353  pCallbackSynchronizationData->m_event.wait(
354  lock,
355  [pCallbackSynchronizationData](){
356  return pCallbackSynchronizationData->state == CallbackState::notified;
357  }
358  );
359  }
360 
361  task();
362 
363  // Notify the waiting CUDA/HIP thread.
364  pCallbackSynchronizationData->state = CallbackState::finished;
365  }
366  pCallbackSynchronizationData->m_event.notify_one();
367 
368 #if BOOST_COMP_HCC // NOTE: workaround for unwanted nonblocking hip streams for HCC (NVCC streams are blocking)
369  queue.m_spQueueImpl->m_callees -= 1;
370 #endif
371  }
372  );
373 
374  t.join();
375  }
376  };
