39  //! The Omp5 memory copy trait.
40  template<
41  typename TDim,
42  typename TViewDst,
43  typename TViewSrc,
44  typename TExtent>
45  struct TaskCopyOmp5
46  {
47  static_assert(
48  !std::is_const<TViewDst>::value,
49  "The destination view can not be const!");
50 
51  static_assert(
52  Dim<TViewSrc>::value == TDim::value,
53  "The source view is required to have dimensionality TDim!");
54  static_assert(
55  Dim<TViewDst>::value == Dim<TViewSrc>::value,
56  "The source and the destination view are required to have the same dimensionality!");
57  static_assert(
58  Dim<TViewDst>::value == Dim<TExtent>::value,
59  "The views and the extent are required to have the same dimensionality!");
60  // TODO: Maybe check for Idx of TViewDst and TViewSrc to have greater or equal range than TExtent.
61  static_assert(
62  std::is_same<Elem<TViewDst>, typename std::remove_const<Elem<TViewSrc>>::type>::value,
63  "The source and the destination view are required to have the same element type!");
64 
65  using Idx = alpaka::Idx<TExtent>;
66 
67  //-----------------------------------------------------------------------------
68  ALPAKA_FN_HOST TaskCopyOmp5(
69  TViewDst & viewDst,
70  TViewSrc const & viewSrc,
71  TExtent const & extent,
72  int const & iDstDevice,
73  int const & iSrcDevice) :
74  m_iDstDevice(iDstDevice),
75  m_iSrcDevice(iSrcDevice),
76  m_extent(castVec<size_t>(extent::getExtentVec(extent))),
77  m_dstPitchBytes(castVec<size_t>(getPitchBytesVec(viewDst))),
78  m_srcPitchBytes(castVec<size_t>(getPitchBytesVec(viewSrc))),
79  m_dstMemNative(reinterpret_cast<void *>(getPtrNative(viewDst))),
80  m_srcMemNative(reinterpret_cast<void const *>(getPtrNative(viewSrc)))
81  {
82 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
83  const auto dstExtent(castVec<size_t>(extent::getExtentVec(viewDst)));
84  const auto srcExtent(castVec<size_t>(extent::getExtentVec(viewSrc)));
85  for(auto i = static_cast<decltype(TDim::value)>(0u); i < TDim::value; ++i)
86  {
87  ALPAKA_ASSERT(m_extent[i] <= dstExtent[i]);
88  ALPAKA_ASSERT(m_extent[i] <= srcExtent[i]);
89  }
90  std::cout << "TaskCopyOmp5<" << TDim::value << ",...>::ctor\tsrcExtent="
91  << srcExtent << ", dstExtent=" << dstExtent << std::endl;
92 #endif
93  }
94 
95 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
96  //-----------------------------------------------------------------------------
97  ALPAKA_FN_HOST auto printDebug() const
98  -> void
99  {
100  std::cout << __func__
101  << " ddev: " << m_iDstDevice
102  << " ew: " << m_extent
103  << " dptr: " << m_dstMemNative
104  << " sdev: " << m_iSrcDevice
105  << " sptr: " << m_srcMemNative
106  << std::endl;
107  }
108 #endif
109  int m_iDstDevice;
110  int m_iSrcDevice;
111  Vec<TDim, size_t> m_extent;
112  Vec<TDim, size_t> m_dstPitchBytes;
113  Vec<TDim, size_t> m_srcPitchBytes;
114  void * m_dstMemNative;
115  void const * m_srcMemNative;
116 
117  //-----------------------------------------------------------------------------
118  //! Executes the kernel function object.
119  ALPAKA_FN_HOST auto operator()() const
120  -> void
121  {
122  ALPAKA_DEBUG_FULL_LOG_SCOPE;
123 
124 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
125  printDebug();
126 #endif
127  constexpr auto lastDim = TDim::value - 1;
128 
129  if(m_extent.prod() > 0)
130  {
131  // offsets == 0 by ptr shift (?)
132  auto dstOffset(Vec<TDim, size_t>::zeros());
133  auto srcOffset(Vec<TDim, size_t>::zeros());
134 
135  auto dstExtentFull(Vec<TDim, size_t>::zeros());
136  auto srcExtentFull(Vec<TDim, size_t>::zeros());
137 
138  const size_t elementSize =
139  ( m_dstPitchBytes[0]%sizeof(Elem<TViewDst>) || m_srcPitchBytes[0]%sizeof(Elem<TViewDst>) )
140  ? 1 : sizeof(Elem<TViewDst>);
141 
142  dstExtentFull[lastDim] = m_dstPitchBytes[lastDim]/elementSize;
143  srcExtentFull[lastDim] = m_srcPitchBytes[lastDim]/elementSize;
144  for(int i = lastDim - 1; i >= 0; --i)
145  {
146  dstExtentFull[i] = m_dstPitchBytes[i]/m_dstPitchBytes[i+1];
147  srcExtentFull[i] = m_srcPitchBytes[i]/m_srcPitchBytes[i+1];
148  }
149 
150  ALPAKA_OMP5_CHECK(
151  omp_target_memcpy_rect(
152  m_dstMemNative, const_cast<void*>(m_srcMemNative),
153  sizeof(Elem<TViewDst>),
154  TDim::value,
155  reinterpret_cast<size_t const *>(&m_extent),
156  reinterpret_cast<size_t const *>(&dstOffset),
157  reinterpret_cast<size_t const *>(&srcOffset),
158  reinterpret_cast<size_t const *>(&dstExtentFull),
159  reinterpret_cast<size_t const *>(&srcExtentFull),
160  m_iDstDevice, m_iSrcDevice));
161  }
162  }
163  };

218 #endif
219  }

234  << " sptr: " << m_srcMemNative
235  << std::endl;
236  }
237 #endif

248 
249  //-----------------------------------------------------------------------------
